centraliza no centro do container

          
              .class{ 
                  display: flex; 
                  justify-content:center;
                }

alinhado ao inicio

            
                .class{ 
                    display: flex;
                    justify-content: flex-start;
                 }

alinhado ao Final

            
                .class{ 
                    display: flex;
                    justify-content: flex-end;
                 }

Separados

            
                .class{ 
                    display: flex;
                    justify-content: space-between;
                 }

Separados uniformemente

            
                .class{ 
                    display: flex;
                    justify-content: space-around;
                }

Separados uniformemente ao redor

            
                .class{ 
                    display: flex;
                    justify-content: space-evenly;
                }

align-items

centraliza no centro do container

            
                .class{ 
                    display: flex; 
                    align-items:center;
                 }

flex-start alinha itens no início do contêiner.

            
                .class{ 
                    display: flex; 
                    align-items:flex-start;
                 }

flex-end alinha itens no final do contêiner.

            
                .class{ 
                    display: flex; 
                    align-items:flex-end;
                 }

baseline alinha itens

            
                .class{ 
                    display: flex; 
                    align-items:baseline;
                 }
